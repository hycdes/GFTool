C++: ADT

string基本方法

#include<string>

string str; char c; char* C;

 

getline(cin,str);  //输入可带空格的一行字符串

str.length();  //返回str长度

str += c;  //直接在str后插入c

str.substr(int p, int n);  //提取str从p往后n长度子串

string &replace(int p, int n, const char* C);  //从p开始的n个字符替换成C[]

string &replace(int p, int n, string &str);  //从p开始的n个字符替换成str

string &replace(int p, int n, string &str, int pos, int num);  //从p开始的n个字符，替换成从str的pos开始的num长度子串

ADT:String
string和数字互相转换

需要 #include<sstream>，整数浮点皆可

字符串转数字（整形为例）：

string str; sstream ss; int n;

ss << str;

ss >> n;

数字转字符串：

ss << n;

str = ss.str();

C++: I/O方法

ios::fixed保留N位小数

#include<iostream>

#include<iomanip>

float f = 1.2;

...

cout.setf(ios::fixed);  //设置按位补0

cout << setprecision(2) << f;  //输出1.20

cout.unsetf(ios::fixed);  //取消按位补0

cout << setprecision(2) << f;  //输出1.2

I/O:iomanip
setfill输出固定位整数

#include<iostream>

#include<iomanip>

int n = 1, m = 123456, s = -1;

...

cout << setw(5) << setfill('0') << n;  //输出00001

cout << setw(5) << setfill('0') << m;  //超过位数如实输出123456

cout << setw(5) << setfill('0') << s;  //输出000-1
